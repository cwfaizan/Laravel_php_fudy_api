POST api/check-account

Example request:
curl --request POST \
    "http://localhost/api/check-account" \
    --header "Content-Type: application/json" \
    --header "Accept: application/json" \
    --data "{
    \"contact_no\": \"923001234567\",
    \"account_type\": \"1 for customer, 2 for waiter\"
}"

const url = new URL(
    "http://localhost/api/check-account"
);

const headers = {
    "Content-Type": "application/json",
    "Accept": "application/json",
};

let body = {
    "contact_no": "923001234567",
    "account_type": "1 for customer, 2 for waiter"
};

fetch(url, {
    method: "POST",
    headers,
    body: JSON.stringify(body),
}).then(response => response.json());

GET api/verify-pin

Example request:
curl --request GET \
    --get "http://localhost/api/verify-pin?item=Use+for+new+account+e.g.+923001234567&user_id=Don%27t+use+for+new+account+e.g.+5&pin_type=14&pin=16" \
    --header "Content-Type: application/json" \
    --header "Accept: application/json"
const url = new URL(
    "http://localhost/api/verify-pin"
);

const params = {
    "item": "Use for new account e.g. 923001234567",
    "user_id": "Don't use for new account e.g. 5",
    "pin_type": "14",
    "pin": "16",
};
Object.keys(params)
    .forEach(key => url.searchParams.append(key, params[key]));

const headers = {
    "Content-Type": "application/json",
    "Accept": "application/json",
};

fetch(url, {
    method: "GET",
    headers,
}).then(response => response.json());
